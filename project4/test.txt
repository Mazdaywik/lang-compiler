
class Coord
    int c

    (int c) ->
        this.c = c

class Point
    Coord coord
    int x
    int y

    (int x, int y) ->
        this.coord = new Coord(5)
        this.coord = null

        this.x = x + 3
        this.y = y + 5


() int main ->
    Point p = new Point(1, 2)
    Point c = null
    c = p
    p = null
    c = p

    int i = 0

    while i < 100000
        p = new Point(1,2)
        c = p
        i++

    return 0
