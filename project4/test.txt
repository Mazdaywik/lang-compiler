
class Point
    int x
    int y

    (int x, int y) ->
        this.x = x
        this.y = y

(Point p) Point incrementPoint ->
    if p == null
        return new Point(3, 2)

    return new Point(p.x + 1, p.y +1 )

(Point p) Point incrementPoint1 ->
    Point c = null

    Point a = null
    Point b = null
    Point k = null

    c = new Point(p.x + 1, p.y +1 )
    a = b
    b = c
    k = c
    a = c

    return c

() int main ->
    Point p = new Point(1, 2)
    Point c = null
    c = p
    p = null
    c = p

    int i = 0

    while i < 10
        p = new Point(i,i+1)

        c = incrementPoint(new Point(i-10,i-20))
        c = p
        incrementPoint(c)
        p = null
        c = null

        c = incrementPoint1(new Point(i-10,i-8))

        incrementPoint1(c)

        c = incrementPoint(c)

        incrementPoint(c)

        p = c
        p = incrementPoint(p)

        p = c
        i++

    return p.x + p.y
